?? Insight del Arquitecto: Cuándo USAR Triggers

Triggers: úsalos con moderación: Los triggers son poderosos pero pueden ser difíciles de debuguear. El arquitecto Martin Kleppmann advierte que los triggers pueden crear efectos secundarios invisibiless. Úsalos para:

  • Auditoría (logging de cambios)
  • Validaciones complejas que no puedes hacer con constraints
  • Mantener datos derivados (ej: timestamp de modificación)
  • Evita usarlos para lógica de negocio compleja (usa procedimientos)

1. ¿Qué es un Trigger?

Trigger: Procedimiento que se ejecuta automáticamente cuando ocurre un evento DML (INSERT, UPDATE, DELETE) en una tabla.

  • BEFORE: Se ejecuta antes de la operación
  • AFTER: Se ejecuta después de la operación
  • FOR EACH ROW: Se ejecuta por cada fila afectada
  • FOR EACH STATEMENT: Se ejecuta una vez por operación

2. Estructura de un Trigger en PostgreSQL

Dos pasos:

  1. Crear la función trigger
  2. Crear el trigger que llama a la función


-- Paso 1: Crear la función trigger

CREATE OR REPLACE FUNCTION nombre_funcion_trigger()

RETURNS TRIGGER AS $$

BEGIN

    -- Lógica del trigger

    -- NEW = nueva fila (para INSERT/UPDATE)

    -- OLD = fila antigua (para UPDATE/DELETE)

    RETURN NEW;  -- o RETURN OLD

END;

$$ LANGUAGE plpgsql;



-- Paso 2: Crear el trigger

CREATE TRIGGER nombre_trigger

    [BEFORE | AFTER] [INSERT | UPDATE | DELETE]

    ON nombre_tabla

    [FOR EACH ROW]

    [WHEN (condición)]

    EXECUTE FUNCTION nombre_funcion_trigger();

3. Variables NEW y OLD

EventoNEWOLD

DELETENULLValores que se eliminarán

Ejemplo: Actualizar fecha de modificación



-- Función para actualizar timestamp

CREATE OR REPLACE FUNCTION actualizar_fecha_mod()

RETURNS TRIGGER AS $$

BEGIN

    NEW.fecha_modificacion := CURRENT_TIMESTAMP;

    RETURN NEW;

END;

$$ LANGUAGE plpgsql;



-- Trigger en tabla producto

CREATE TRIGGER trigger_fecha_mod_producto

    BEFORE UPDATE ON producto

    FOR EACH ROW

    EXECUTE FUNCTION actualizar_fecha_mod();

4. Validaciones con Triggers



-- Evitar que el precio baje

CREATE OR REPLACE FUNCTION validar_precio()

RETURNS TRIGGER AS $$

BEGIN

    IF NEW.precio < OLD.precio THEN

        RAISE EXCEPTION 'No se puede reducir el precio de % a %', 

            OLD.precio, NEW.precio;

    END IF;

    RETURN NEW;

END;

$$ LANGUAGE plpgsql;



CREATE TRIGGER trigger_validar_precio

    BEFORE UPDATE ON producto

    FOR EACH ROW

    EXECUTE FUNCTION validar_precio();



-- Validar que el stock no sea negativo

CREATE OR REPLACE FUNCTION validar_stock()

RETURNS TRIGGER AS $$

BEGIN

    IF NEW.cantidad < 0 THEN

        RAISE EXCEPTION 'La cantidad no puede ser negativa';

    END IF;

    RETURN NEW;

END;

$$ LANGUAGE plpgsql;

5. Auditoría con Triggers



-- Tabla de auditoría

CREATE TABLE auditoria_cliente (

    id SERIAL PRIMARY KEY,

    usuario VARCHAR(100),

    accion VARCHAR(10),

    cliente_id INTEGER,

    datos_anteriores TEXT,

    datos_nuevos TEXT,

    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP

);



-- Función de auditoría

CREATE OR REPLACE FUNCTION auditar_cliente()

RETURNS TRIGGER AS $$

BEGIN

    IF TG_OP = 'INSERT' THEN

        INSERT INTO auditoria_cliente (usuario, accion, cliente_id, datos_nuevos)

        VALUES (current_user, 'INSERT', NEW.id, row_to_json(NEW)::text);

        RETURN NEW;

    ELSIF TG_OP = 'UPDATE' THEN

        INSERT INTO auditoria_cliente (usuario, accion, cliente_id, datos_anteriores, datos_nuevos)

        VALUES (current_user, 'UPDATE', NEW.id, row_to_json(OLD)::text, row_to_json(NEW)::text);

        RETURN NEW;

    ELSIF TG_OP = 'DELETE' THEN

        INSERT INTO auditoria_cliente (usuario, accion, cliente_id, datos_anteriores)

        VALUES (current_user, 'DELETE', OLD.id, row_to_json(OLD)::text);

        RETURN OLD;

    END IF;

    RETURN NULL;

END;

$$ LANGUAGE plpgsql;



-- Triggers de auditoría

CREATE TRIGGER trigger_auditar_cliente_insert

    AFTER INSERT ON cliente

    FOR EACH ROW

    EXECUTE FUNCTION auditar_cliente();



CREATE TRIGGER trigger_auditar_cliente_update

    AFTER UPDATE ON cliente

    FOR EACH ROW

    EXECUTE FUNCTION auditar_cliente();



CREATE TRIGGER trigger_auditar_cliente_delete

    AFTER DELETE ON cliente

    FOR EACH ROW

    EXECUTE FUNCTION auditar_cliente();

6. Triggers con Condición WHEN



-- Solo dispara cuando se cumple la condición

CREATE TRIGGER trigger_precio_alto

    AFTER UPDATE ON producto

    FOR EACH ROW

    WHEN (NEW.precio > 1000)  -- solo si precio > 1000

    EXECUTE FUNCTION notificar_precio_alto();



-- Función asociada

CREATE OR REPLACE FUNCTION notificar_precio_alto()

RETURNS TRIGGER AS $$

BEGIN

    RAISE NOTICE 'Producto % tiene nuevo precio alto: %', 

        NEW.nombre, NEW.precio;

    RETURN NEW;

END;

$$ LANGUAGE plpgsql;

7. Eventos de Triggers

EventoDescripción
INSERTCuando se inserta una fila
UPDATECuando se actualiza una fila
DELETECuando se elimina una fila
TRUNCATECuando se trunca la tabla

?? Notas importantes:

  • En PostgreSQL, un trigger NO puede modificar otras tablas directamente (salvo que llame a funciones)
  • Usa RETURN NEW (o OLD para DELETE) al final de la función
  • TG_OP indica la operación que disparó el trigger
  • current_user indica el usuario que ejecutó la operación
