Ejercicio 1: Funciones

Crea una funciˇn que reciba un rango de fechas y retorne el total de ventas en ese perÝodo.

CREATE OR REPLACE FUNCTION ventas_por_periodo(
    p_fecha_inicio DATE,
    p_fecha_fin DATE
)
RETURNS NUMERIC AS $$
DECLARE
    v_total NUMERIC := 0;
BEGIN
    SELECT COALESCE(SUM(total), 0)
    INTO v_total
    FROM pedido
    WHERE fecha BETWEEN p_fecha_inicio AND p_fecha_fin;
    
    RETURN v_total;
END;
$$ LANGUAGE plpgsql;

Ejercicio 2: Procedimiento

Crea un procedimiento que transfiera productos entre almacenes.

CREATE OR REPLACE PROCEDURE transferir_producto(
    p_producto_id INTEGER,
    p_origen_id INTEGER,
    p_destino_id INTEGER,
    p_cantidad INTEGER
) AS $$
BEGIN
    -- Verificar stock en origen
    UPDATE inventario
    SET cantidad = cantidad - p_cantidad
    WHERE producto_id = p_producto_id 
      AND almacen_id = p_origen_id;
    
    -- Agregar al destino
    UPDATE inventario
    SET cantidad = cantidad + p_cantidad
    WHERE producto_id = p_producto_id 
      AND almacen_id = p_destino_id;
    
    RAISE NOTICE 'Transferencia completada: % unidades', p_cantidad;
END;
$$ LANGUAGE plpgsql;

Ejercicio 3: Trigger de auditorÝa

Crea un trigger que registre cambios en la tabla producto.

CREATE TABLE historial_producto (
    id SERIAL PRIMARY KEY,
    producto_id INTEGER,
    accion VARCHAR(10),
    campo_modificado VARCHAR(50),
    valor_anterior TEXT,
    valor_nuevo TEXT,
    usuario VARCHAR(50),
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE OR REPLACE FUNCTION auditar_producto()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'UPDATE' THEN
        IF OLD.precio <> NEW.precio THEN
            INSERT INTO historial_producto 
                (producto_id, accion, campo_modificado, valor_anterior, valor_nuevo, usuario)
            VALUES (NEW.id, 'UPDATE', 'precio', OLD.precio::text, NEW.precio::text, current_user);
        END IF;
        IF OLD.stock <> NEW.stock THEN
            INSERT INTO historial_producto 
                (producto_id, accion, campo_modificado, valor_anterior, valor_nuevo, usuario)
            VALUES (NEW.id, 'UPDATE', 'stock', OLD.stock::text, NEW.stock::text, current_user);
        END IF;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_auditar_producto
    AFTER UPDATE ON producto
    FOR EACH ROW
    EXECUTE FUNCTION auditar_producto();
