Insight del Arquitecto: PL/SQL como Herramienta de Poder

Los subprogramas almacenados (procedimientos, funciones, paquetes) son el corazon de la logica de negocio en la base de datos. Mientras que el SQL puro es declarativo, PL/SQL permite programacion imperativa: condiciones, loops, manejo de excepciones. Un buen arquitecto sabe cuando usar cada uno. Los triggers, por otro lado, deben usarse con cuidado -- un trigger mal disenado puede destruir el rendimiento de una base de datos completa.

1. Cursores en PL/SQL

Recorrido de filas

Un cursor permite recuperar filas de una consulta SELECT una por una y procesarlas individualmente.

Cursor Manual con FETCH

DECLARE
    CURSOR ejecursor IS
        SELECT nombre, salario FROM empleado
        WHERE salario < 5000;
    v_nombre empleado.nombre%TYPE;
    v_salario empleado.salario%TYPE;
BEGIN
    OPEN ejecursor;
    LOOP
        FETCH ejecursor INTO v_nombre, v_salario;
        EXIT WHEN ejecursor%NOTFOUND;
        INSERT INTO TEMP VALUES (v_nombre, v_salario);
    END LOOP;
    CLOSE ejecursor;
END;

Cursor con FOR LOOP (Automatico)

DECLARE
    CURSOR ejecursor IS
        SELECT nombre, salario FROM empleado
        WHERE salario < 5000;
BEGIN
    FOR ejecursor_rec IN ejecursor LOOP
        INSERT INTO TEMP
        VALUES (ejecursor_rec.nombre, ejecursor_rec.salario);
    END LOOP;
END;

Atributos del Cursor

Atributo Tipo Descripcion
%ISOPEN Booleano TRUE si el cursor esta abierto
%FOUND Booleano TRUE mientras el ultimo FETCH trae una fila
%NOTFOUND Booleano TRUE cuando el ultimo FETCH no trae filas
%ROWCOUNT Numerico Total de filas recuperadas hasta el momento

2. Procedimientos Almacenados

CREATE OR REPLACE PROCEDURE

Un procedimiento ejecuta una accion. Puede tener parametros de entrada (IN), salida (OUT) o ambos (IN OUT).

Sintaxis general:

CREATE OR REPLACE PROCEDURE <nombre>
    [(parametro, ...)]
IS
    -- declaraciones
BEGIN
    -- logica
    [EXCEPTION]
END;

Ejemplo 1: Actualizar Salario

Procedimiento con parametro IN: Actualiza el salario de un empleado por su ID.

CREATE OR REPLACE PROCEDURE cambio_salario
    (v_id_emp IN NUMBER,
     v_nuevo_salario IN NUMBER)
IS
BEGIN
    UPDATE empleado
    SET salario = v_nuevo_salario
    WHERE cod_empleado = v_id_emp;
    COMMIT;
END cambio_salario;

-- Ejecucion:
EXECUTE cambio_salario(7839, 12000);

Ejemplo 2: Con Manejo de Excepciones

Procedimiento con validacion: Incrementa el salario si existe, lanza excepcion si es nulo.

CREATE OR REPLACE PROCEDURE actualiza_salario
    (v_id_emp NUMBER, incremento NUMBER)
IS
    salario_actual NUMBER;
    salario_olvidado EXCEPTION;
BEGIN
    SELECT salary INTO salario_actual
    FROM employees
    WHERE employee_id = v_id_emp;

    IF salario_actual IS NULL THEN
        RAISE salario_olvidado;
    ELSE
        UPDATE employees
        SET salary = salary + incremento
        WHERE employee_id = v_id_emp;
    END IF;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        INSERT INTO emp_audit
        VALUES (v_id_emp, 'Empleado no existe');
    WHEN salario_olvidado THEN
        INSERT INTO emp_audit
        VALUES (v_id_emp, 'Salario es nulo');
END actualiza_salario;

3. Funciones

RETURN valor

Una funcion es similar a un procedimiento, pero obligatoriamente retorna un valor al entorno llamante mediante la clausula RETURN.

Sintaxis general:

CREATE OR REPLACE FUNCTION <nombre>
    [(parametro, ...)]
RETURN tipo_dato
IS
    -- declaraciones
BEGIN
    -- logica
    RETURN valor;
    [EXCEPTION]
END;

Ejemplo 1: Calculo de Impuesto

CREATE OR REPLACE FUNCTION impuesto
    (v_valor IN NUMBER)
RETURN NUMBER
IS
BEGIN
    RETURN (v_valor * 0.30);
END impuesto;

-- Uso en consulta:
SELECT salary, impuesto(salary) AS IMPUESTO
FROM employees
WHERE employee_id = 100;

Ejemplo 2: Validacion de Salario

CREATE OR REPLACE FUNCTION salario_ok
    (v_salario NUMBER, v_jobid NUMBER)
RETURN NUMBER
IS
    min_sal NUMBER;
    max_sal NUMBER;
BEGIN
    SELECT min_salary, max_salary
    INTO min_sal, max_sal
    FROM jobs
    WHERE job_id = v_jobid;

    IF (v_salario >= min_sal) AND (v_salario <= max_sal) THEN
        RETURN 1;
    ELSE
        RETURN 0;
    END IF;
END salario_ok;

4. Triggers (Disparadores)

BEFORE / AFTER · FOR EACH ROW

Un trigger es un bloque PL/SQL que se ejecuta automaticamente cuando ocurre un evento DML (INSERT, UPDATE, DELETE) sobre una tabla.

Sintaxis general:

CREATE OR REPLACE TRIGGER <nombre>
    BEFORE|AFTER INSERT|UPDATE|DELETE [OF columna] ON tabla
    FOR EACH ROW
    [WHEN condicion]
DECLARE
    -- variables
BEGIN
    -- logica
    -- :NEW.columna  -- valor nuevo
    -- :OLD.columna  -- valor anterior
END;

Ejemplo: Validacion de Rango Salarial

Trigger BEFORE: Verifica que el salario nuevo este dentro del rango del cargo y que el incremento no exceda el 10%.

CREATE OR REPLACE TRIGGER chequea_salario
    BEFORE INSERT OR UPDATE OF salary, job_id ON employees
    FOR EACH ROW
    WHEN (new.job_id != 'AD_PRES')
DECLARE
    minsal NUMBER;
    maxsal NUMBER;
BEGIN
    SELECT min_salary, max_salary
    INTO minsal, maxsal
    FROM jobs
    WHERE job_id = :new.job_id;

    IF (:new.salary < minsal OR :new.salary > maxsal) THEN
        raise_application_error(-20225, 'Salario fuera de rango');
    ELSIF (:new.salary < :old.salary) THEN
        raise_application_error(-20320, 'Incremento negativo');
    ELSIF (:new.salary > 1.1 * :old.salary) THEN
        raise_application_error(-20325, 'Incremento excede el 10%');
    END IF;
END chequea_salario;

5. Paquetes (Packages)

Agrupacion de subprogramas

Un paquete agrupa logicamente procedimientos y funciones relacionados. Tiene dos partes: la especificacion (interfaz publica) y el cuerpo (implementacion).

Estructura:

-- Especificacion (interfaz publica)
CREATE PACKAGE emp_acciones AS
    PROCEDURE contrata_empleado(
        ename VARCHAR2, job VARCHAR2, mgr NUMBER,
        catpago NUMBER, sal NUMBER, deptno NUMBER);
    PROCEDURE despide_empleado(emp_id NUMBER);
END emp_acciones;

-- Cuerpo (implementacion)
CREATE PACKAGE BODY emp_acciones AS
    PROCEDURE contrata_empleado(
        ename VARCHAR2, job VARCHAR2, mgr NUMBER,
        catpago NUMBER, sal NUMBER, deptno NUMBER) IS
    BEGIN
        INSERT INTO empleado
        VALUES (empno_seq.NEXTVAL, ename, job, mgr,
                SYSDATE, catpago, sal, deptno);
    END contrata_empleado;

    PROCEDURE despide_empleado(emp_id NUMBER) IS
    BEGIN
        DELETE FROM empleado
        WHERE cod_empleado = emp_id;
    END despide_empleado;
END emp_acciones;

6. Variables de Sustitucion

& y && en SQL*Plus

SQL*Plus permite usar variables de sustitucion para ingresar valores en tiempo de ejecucion.

& (un solo uso): Pide el valor cada vez que aparece.

SELECT nombre
FROM empleado
WHERE cod_empleado = &cod_empleado;

&& (reutilizable): Pide el valor solo la primera vez y lo reutiliza.

SELECT nombre, &&nombre_columna
FROM empleado
ORDER BY &nombre_columna;
