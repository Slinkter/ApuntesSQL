1. Roles y Usuarios

-- Crear rol (sin login)
CREATE ROLE analyst;
CREATE ROLE developer;

-- Crear usuario (con login)
CREATE USER juan WITH PASSWORD 'secret123';
CREATE USER maria LOGIN PASSWORD 'pass456';

-- Modificar rol
ALTER ROLE juan WITH SUPERUSER;
ALTER ROLE developer SET default_transaction_isolation = 'read committed';

-- Eliminar
DROP ROLE analyst;
DROP USER juan;

2. Permisos con GRANT/REVOKE

-- Permisos en tabla
GRANT SELECT ON empleados TO analyst;
GRANT SELECT, INSERT, UPDATE ON empleados TO developer;
GRANT ALL ON empleados TO admin;

-- Permisos en esquema
GRANT USAGE ON SCHEMA public TO analyst;
GRANT ALL ON ALL TABLES IN SCHEMA public TO developer;

-- Permisos en funciones
GRANT EXECUTE ON FUNCTION calculate_bonus() TO developer;

-- Quitar permisos
REVOKE INSERT ON empleados FROM developer;

3. Row Level Security (RLS)

-- Habilitar RLS en tabla
ALTER TABLE pedidos ENABLE ROW LEVEL SECURITY;

-- PolÃ­tica: usuarios ven solo sus pedidos
CREATE POLICY mis_pedidos ON pedidos
    USING (id_cliente = current_user);

-- PolÃ­tica por rol
CREATE POLICY pedidos_gerente ON pedidos
    FOR ALL
    USING (
        EXISTS (
            SELECT 1 FROM usuarios_roles
            WHERE usuario = current_user AND rol = 'gerente'
        )
    );

4. Esquemas

-- Crear esquema
CREATE SCHEMA ventas;

-- Mover tabla a esquema
ALTER TABLE pedidos SET SCHEMA ventas;

-- Acceso a esquema
GRANT USAGE ON SCHEMA ventas TO analyst;
