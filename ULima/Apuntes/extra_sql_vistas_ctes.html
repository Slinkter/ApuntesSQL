?? Insight del Arquitecto: Optimización de JOINs

PostgreSQL y los JOINs: El optimizador de PostgreSQL es muy bueno eligiendo el método de JOIN (Nested Loop, Hash Join, Merge Join). El orden de las tablas en el FROM importa menos que las condiciones en el ON. Usa EXPLAIN ANALYZE para ver el plan de ejecución real.


EXPLAIN ANALYZE 

SELECT * FROM cliente c 

JOIN pedido p ON c.id = p.cliente_id 

WHERE p.fecha > '2024-01-01';

1. Tipos de JOIN

JOIN: Combina filas de dos o más tablas basándose en una condición relacionada.

INNER JOIN (Equijoin)


-- Sintaxis estándar (preferida en PostgreSQL)

SELECT c.nombre, p.numero_orden

FROM cliente c

INNER JOIN pedido p ON c.id = p.cliente_id;



-- Sintaxis antigua (Oracle, SQL Server legacy)

SELECT c.nombre, p.numero_orden

FROM cliente c, pedido p

WHERE c.id = p.cliente_id;



-- Solo devuelve clientes que tienen pedidos

LEFT OUTER JOIN


-- Devuelve TODOS los clientes, con sus pedidos (si existen)

SELECT c.nombre, p.numero_orden

FROM cliente c

LEFT JOIN pedido p ON c.id = p.cliente_id;



-- Los clientes SIN pedidos mostrarán NULL en columnas de pedido

RIGHT OUTER JOIN


-- Devuelve TODOS los pedidos, con su cliente (si existe)

SELECT c.nombre, p.numero_orden

FROM cliente c

RIGHT JOIN pedido p ON c.id = p.cliente_id;

CROSS JOIN


-- Producto cartesiano (todas las combinaciones)

-- Útil para generar datos de prueba

SELECT c.nombre, p.nombre_producto

FROM cliente c

CROSS JOIN producto p;



-- Equivalente a (evitar en producción):

SELECT c.nombre, p.nombre_producto

FROM cliente c, producto p;

SELF JOIN


-- Une una tabla consigo misma

-- Ej: Encontrar empleados que trabajan para el mismo jefe



SELECT e1.nombre AS empleado, 

       e2.nombre AS jefe

FROM empleado e1

JOIN empleado e2 ON e1.jefe_id = e2.id;



-- Ej: Todos los empleados del mismo departamento

SELECT a.nombre, b.nombre

FROM empleado a

JOIN empleado b ON a.departamento_id = b.departamento_id

WHERE a.id <> b.id;

2. Non-Equi JOIN


-- JOIN sin igualdad (usando rangos)

-- Ej: Categorizar productos por precio



SELECT p.nombre, p.precio, c.categoria

FROM producto p

JOIN categoria_precio c 

    ON p.precio BETWEEN c.precio_min AND c.precio_max;

3. Subconsultas

Subconsulta: Consulta anidada dentro de otra consulta. Se ejecuta primero y su resultado se usa en la consulta principal.

Subconsulta en WHERE


-- Subconsulta de fila simple (=, <, >, etc.)

SELECT * FROM pedido

WHERE cliente_id = (

    SELECT id FROM cliente 

    WHERE nombre = 'JUAN'

);



-- Subconsulta de múltiples filas (IN, ANY, ALL)

SELECT * FROM producto

WHERE id IN (

    SELECT id_producto FROM detalle_pedido 

    WHERE cantidad > 10

);



-- Con función de agregación

SELECT * FROM producto

WHERE precio > (

    SELECT AVG(precio) FROM producto

);

Subconsulta Correlacionada


-- La subconsulta referencia la consulta principal

SELECT p.nombre, p.precio

FROM producto p

WHERE p.precio > (

    SELECT AVG(precio) 

    FROM producto p2 

    WHERE p2.categoria_id = p.categoria_id

);

Subconsulta en FROM (Tabla derivada)


SELECT sub.total_pedidos, sub.cliente_nombre

FROM (

    SELECT c.nombre AS cliente_name, 

           COUNT(p.id) AS total_pedidos

    FROM cliente c

    LEFT JOIN pedido p ON c.id = p.cliente_id

    GROUP BY c.nombre

) AS sub

WHERE sub.total_pedidos > 5;

Subconsulta con EXISTS


-- Clientes que tienen al menos un pedido

SELECT * FROM cliente c

WHERE EXISTS (

    SELECT 1 FROM pedido p 

    WHERE p.cliente_id = c.id

);



-- Clientes SIN pedidos (negación)

SELECT * FROM cliente c

WHERE NOT EXISTS (

    SELECT 1 FROM pedido p 

    WHERE p.cliente_id = c.id

);

4. Funciones de Grupo con GROUP BY


-- Promedio por categoría

SELECT categoria_id, 

       AVG(precio) AS precio_promedio,

       COUNT(*) AS productos,

       SUM(stock) AS stock_total

FROM producto

GROUP BY categoria_id;



-- Con filtro de filas (WHERE) antes de agrupar

SELECT categoria_id, AVG(precio) AS promedio

FROM producto

WHERE activo = true

GROUP BY categoria_id;



-- Con filtro de grupos (HAVING) después de agrupar

SELECT cliente_id, COUNT(*) AS num_pedidos

FROM pedido

GROUP BY cliente_id

HAVING COUNT(*) > 3;

5. Ejercicios Completos

Ejercicios:

  1. Lista clientes con sus pedidos (incluye clientes sin pedidos)
  2. Encuentra el producto más vendido
  3. Lista el total gastado por cada cliente
  4. Muestra los productos que nunca se han vendido
  5. Encuentra el segundo producto más caro

-- Soluciones:



-- 1. Clientes con pedidos (LEFT JOIN)

SELECT c.nombre, COALESCE(COUNT(p.id), 0) AS pedidos

FROM cliente c

LEFT JOIN pedido p ON c.id = p.cliente_id

GROUP BY c.nombre;



-- 2. Producto más vendido

SELECT p.nombre, SUM(d.cantidad) AS total_vendido

FROM producto p

JOIN detalle_pedido d ON p.id = d.producto_id

GROUP BY p.nombre

ORDER BY total_vendido DESC

LIMIT 1;



-- 4. Productos nunca vendidos

SELECT p.nombre

FROM producto p

WHERE NOT EXISTS (

    SELECT 1 FROM detalle_pedido d 

    WHERE d.producto_id = p.id

);



-- 5. Segundo producto más caro

SELECT nombre, precio

FROM producto

ORDER BY precio DESC

LIMIT 1 OFFSET 1;
