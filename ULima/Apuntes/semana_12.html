1. Cursores

-- Cursor bГѓВЎsico
CREATE OR REPLACE FUNCTION procesar_empleados()
RETURNS TEXT AS $$
DECLARE
    emp_record RECORD;
    cur_emp CURSOR FOR SELECT id, nombre, salary FROM employees;
    total_salary NUMERIC := 0;
BEGIN
    OPEN cur_emp;
    LOOP
        FETCH cur_emp INTO emp_record;
        EXIT WHEN NOT FOUND;
        total_salary := total_salary + emp_record.salary;
    END LOOP;
    CLOSE cur_emp;
    RETURN 'Total: ' || total_salary;
END;
$$ LANGUAGE plpgsql;

-- Cursor con parГѓВЎmetro
CREATE OR REPLACE FUNCTION buscar_empleado(p_dept VARCHAR)
RETURNS TABLE(id INTEGER, nombre TEXT) AS $$
DECLARE
    cur CURSOR(p_dept VARCHAR) FOR 
        SELECT id, nombre FROM employees WHERE department = p_dept;
BEGIN
    FOR emp IN cur(p_dept) LOOP
        RETURN NEXT emp;
    END LOOP;
END;
$$ LANGUAGE plpgsql;

2. Triggers

-- FunciГѓВіn de trigger
CREATE OR REPLACE FUNCTION fn_audit_insert()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO audit_log (tabla, operacion, usuario, fecha)
    VALUES (TG_TABLE_NAME, TG_OP, current_user, now());
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Crear trigger
CREATE TRIGGER tr_empleados_insert
AFTER INSERT ON empleados
FOR EACH ROW
EXECUTE FUNCTION fn_audit_insert();

-- Trigger BEFORE (modificar antes de insertar)
CREATE OR REPLACE FUNCTION fn_validar_salary()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.salary < 0 THEN
        RAISE EXCEPTION 'Salary no puede ser negativo';
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER tr_validar_salary
BEFORE INSERT OR UPDATE ON employees
FOR EACH ROW
EXECUTE FUNCTION fn_validar_salary();

3. Manejo de Excepciones

CREATE OR REPLACE FUNCTION division_segura(a NUMERIC, b NUMERIC)
RETURNS NUMERIC AS $$
DECLARE
    resultado NUMERIC;
BEGIN
    resultado := a / b;
    RETURN resultado;
EXCEPTION
    WHEN division_by_zero THEN
        RAISE NOTICE 'DivisiГѓВіn por cero';
        RETURN NULL;
    WHEN OTHERS THEN
        RAISE NOTICE 'Error: %', SQLERRM;
        RETURN -1;
END;
$$ LANGUAGE plpgsql;

EJERCICIO

Crear trigger que registre en log cuando se elimine un empleado.

CREATE OR REPLACE FUNCTION log_delete_empleado()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO log_eliminados (nombre, fecha)
    VALUES (OLD.nombre, now());
    RETURN OLD;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER tr_empleado_delete
AFTER DELETE ON empleados
FOR EACH ROW
EXECUTE FUNCTION log_delete_empleado();
