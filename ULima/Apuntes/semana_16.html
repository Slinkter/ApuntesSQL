3. Consulta SQL Completa de Ejemplo

-- Consulta compleja: Top clientes con más pedidos y mayor gasto
SELECT 
    c.id,
    c.nombre,
    c.email,
    COUNT(p.id) AS total_pedidos,
    COALESCE(SUM(p.total), 0) AS gasto_total,
    COALESCE(AVG(p.total), 0) AS promedio_pedido,
    MAX(p.fecha) AS ultimo_pedido
FROM cliente c
LEFT JOIN pedido p ON c.id = p.cliente_id
WHERE c.activo = true
GROUP BY c.id, c.nombre, c.email
HAVING COUNT(p.id) > 0
ORDER BY gasto_total DESC
LIMIT 10;

4. Funciones PL/pgSQL de Repaso

-- Función con manejo de errores
CREATE OR REPLACE FUNCTION buscar_cliente(p_nombre VARCHAR)
RETURNS TABLE(id INTEGER, nombre VARCHAR) AS $$
BEGIN
    RETURN QUERY 
        SELECT c.id, c.nombre 
        FROM cliente c 
        WHERE c.nombre ILIKE '%' || p_nombre || '%';
EXCEPTION
    WHEN OTHERS THEN
        RAISE NOTICE 'Error en búsqueda: %', SQLERRM;
        RETURN;
END;
$$ LANGUAGE plpgsql;

5. Trigger de Auditoría Completo

-- Tabla de auditoría
CREATE TABLE log_cambios (
    id SERIAL PRIMARY KEY,
    tabla VARCHAR(50),
    operacion VARCHAR(10),
    registro_id INTEGER,
    usuario VARCHAR(50),
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    datos JSONB
);

-- Función trigger genérica
CREATE OR REPLACE FUNCTION auditar_cambio()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        INSERT INTO log_cambios (tabla, operacion, registro_id, usuario, datos)
        VALUES (TG_TABLE_NAME, 'INSERT', NEW.id, current_user, to_jsonb(NEW));
        RETURN NEW;
    ELSIF TG_OP = 'UPDATE' THEN
        INSERT INTO log_cambios (tabla, operacion, registro_id, usuario, datos)
        VALUES (TG_TABLE_NAME, 'UPDATE', NEW.id, current_user, to_jsonb(NEW));
        RETURN NEW;
    ELSIF TG_OP = 'DELETE' THEN
        INSERT INTO log_cambios (tabla, operacion, registro_id, usuario, datos)
        VALUES (TG_TABLE_NAME, 'DELETE', OLD.id, current_user, to_jsonb(OLD));
        RETURN OLD;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- Aplicar a cualquier tabla
CREATE TRIGGER audit_cliente
    AFTER INSERT OR UPDATE OR DELETE ON cliente
    FOR EACH ROW EXECUTE FUNCTION auditar_cambio();
