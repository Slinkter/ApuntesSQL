?? Insight del Arquitecto: PL/pgSQL vs PL/SQL

PostgreSQL usa PL/pgSQL (no PL/SQL de Oracle):

  • Sintaxis similar pero no idéntica
  • Funciones con RETURN, no RETURNING (en funciones)
  • CREATE OR REPLACE funciona para funciones
  • Soporta triggers con eventos específicos
  • RETURN NEXT para retornar conjuntos

4. Funciones en PostgreSQL


-- Función simple

CREATE OR REPLACE FUNCTION calcular_igv(precio NUMERIC)

RETURNS NUMERIC AS $$

BEGIN

    RETURN precio * 0.18;

END;

$$ LANGUAGE plpgsql;



-- Función con return de tabla

CREATE OR REPLACE FUNCTION get_clientes_activos()

RETURNS TABLE(

    id INTEGER,

    nombre VARCHAR(100)

) AS $$

BEGIN

    RETURN QUERY 

        SELECT c.id, c.nombre 

        FROM cliente c 

        WHERE c.activo = true;

END;

$$ LANGUAGE plpgsql;



-- Función con manejo de errores

CREATE OR REPLACE FUNCTION dividir(a NUMERIC, b NUMERIC)

RETURNS NUMERIC AS $$

BEGIN

    IF b = 0 THEN

        RAISE EXCEPTION 'División por cero no permitida';

    END IF;

    RETURN a / b;

EXCEPTION

    WHEN OTHERS THEN

        RAISE NOTICE 'Error: %', SQLERRM;

        RETURN NULL;

END;

$$ LANGUAGE plpgsql;

5. Procedimientos (CREATE PROCEDURE)


-- Procedimiento para actualizar stock

CREATE OR REPLACE PROCEDURE actualizar_stock(

    p_producto_id INTEGER,

    p_cantidad INTEGER

) AS $$

BEGIN

    UPDATE producto

    SET stock = stock + p_cantidad

    WHERE id = p_producto_id;

    

    RAISE NOTICE 'Stock actualizado para producto %', p_producto_id;

END;

$$ LANGUAGE plpgsql;



-- Llamar procedimiento

CALL actualizar_stock(10, 5);

6. Cursores


-- Cursor explícito

DECLARE

    c_cliente CURSOR FOR 

        SELECT nombre, email FROM cliente 

        WHERE activo = true;

    v_registro RECORD;

BEGIN

    OPEN c_cliente;

    LOOP

        FETCH c_cliente INTO v_registro;

        EXIT WHEN NOT FOUND;

        RAISE NOTICE 'Cliente: % - %', 

            v_registro.nombre, v_registro.email;

    END LOOP;

    CLOSE c_cliente;

END;



-- Cursor con parámetros

DECLARE

    c_pedidos CURSOR(p_cliente_id INTEGER) FOR

        SELECT * FROM pedido 

        WHERE cliente_id = p_cliente_id;

BEGIN

    -- Abrir con parámetro

    OPEN c_pedidos(5);

    -- ...

END;

8. Errores y Excepciones


-- Tipos de RAISE

RAISE NOTICE 'Mensaje informativo';

RAISE WARNING 'Advertencia';

RAISE EXCEPTION 'Error fatal';



-- Con variables

RAISE NOTICE 'El cliente % tiene % pedidos', v_nombre, v_contador;



-- Manejo de excepciones

BEGIN

    -- operación que puede fallar

EXCEPTION

    WHEN NO_DATA_FOUND THEN

        RAISE NOTICE 'No se encontró el registro';

    WHEN TOO_MANY_ROWS THEN

        RAISE NOTICE 'Se devolvieron múltiples filas';

    WHEN OTHERS THEN

        RAISE NOTICE 'Error: %', SQLERRM;

END;
