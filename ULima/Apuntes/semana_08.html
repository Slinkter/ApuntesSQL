Seccion 4: SQL - Soluciones

Ejercicio 1 - DDL:

CREATE TABLE departamento (
    cod_depto CHAR(3) PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL,
    presupuesto DECIMAL(10,2) CHECK (presupuesto > 0)
);

CREATE TABLE empleado (
    cod_emp INT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    salario DECIMAL(8,2),
    cod_depto CHAR(3),
    FOREIGN KEY (cod_depto) REFERENCES departamento(cod_depto)
    ON DELETE CASCADE
);

Ejercicio 2 - INSERT:

BEGIN;

INSERT INTO departamento VALUES
    ('TI', 'Tecnologia Informatica', 50000.00),
    ('VEN', 'Ventas Corporativas', 35000.00);

INSERT INTO empleado (cod_emp, nombre, salario, cod_depto) VALUES
    (1, 'Ana Torres', 4500.00, 'TI'),
    (2, 'Luis Gomez', 3200.00, 'TI'),
    (3, 'Pedro Diaz', 1800.00, 'VEN');

COMMIT;

Ejercicio 3 - UPDATE:

UPDATE empleado
SET salario = salario * 1.15
WHERE cod_depto = (SELECT cod_depto FROM departamento WHERE nombre = 'TI');

Ejercicio 4 - DELETE:

DELETE FROM empleado
WHERE salario < 1500
  AND cod_depto = (SELECT cod_depto FROM departamento WHERE nombre = 'VENTAS');

Ejercicio 5 - SELECT con JOIN:

SELECT e.nombre AS empleado,
       d.nombre AS departamento,
       p.nombre AS proyecto
FROM empleado e
JOIN departamento d ON e.cod_depto = d.cod_depto
LEFT JOIN proyecto p ON d.cod_depto = p.cod_depto
WHERE e.salario > (SELECT AVG(salario) FROM empleado)
ORDER BY e.salario DESC;
