-- 2. Clientes sin pedidos en los ˙ltimos 6 meses

FROM cliente c
WHERE NOT EXISTS (
    SELECT 1 FROM pedido p 
    WHERE p.cliente_id = c.id 
    AND p.fecha > CURRENT_DATE - INTERVAL '6 months'
);

-- 3. Top 5 productos por ingresos
SELECT p.nombre, SUM(d.cantidad * d.precio) as ingresos
FROM producto p
JOIN detalle_pedido d ON p.id = d.producto_id
GROUP BY p.nombre
ORDER BY ingresos DESC
LIMIT 5;
