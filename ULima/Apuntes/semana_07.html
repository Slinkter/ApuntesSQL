3. Subconsultas

-- Subconsulta escalar (retorna un valor)
SELECT nombre, salary,
    (SELECT AVG(salary) FROM empleados) as promedio
FROM empleados;

-- Subconsulta de tabla (retorna conjunto)
SELECT nombre
FROM empleados
WHERE department_id IN (
    SELECT id FROM departments WHERE nombre = 'Ventas'
);

-- Subconsulta en FROM (tabla derivada)
SELECT dept, total
FROM (
    SELECT department as dept, COUNT(*) as total
    FROM empleados
    GROUP BY department
) as resumen
WHERE total > 5;
