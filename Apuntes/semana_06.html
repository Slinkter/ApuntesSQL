1. Funciones para Valores Nulos

COALESCE(valor1, valor2, ...)

Retorna el primer valor no nulo de la lista.

-- Si telefono es NULL, muestra 'No disponible'
SELECT nombre, COALESCE(telefono, 'No disponible') as contacto
FROM clientes;

-- Común:默认值 si falta dato
SELECT nombre, COALESCE(apellido, '') || ' ' || COALESCE(nombre, '') as nombre_completo
FROM empleados;

NULLIF(valor1, valor2)

Retorna NULL si valor1 = valor2. Útil para evitar división por cero.

-- Si division es 0, retorna NULL (evita error)
SELECT precio / NULLIF(cantidad, 0) as precio_unitario
FROM productos;

2. CASE - Expresiones Condicionales

-- CASE simple
SELECT nombre, 
    CASE departamento
        WHEN 'Ventas' THEN 'Comercial'
        WHEN 'TI' THEN 'Tecnología'
        WHEN 'RRHH' THEN 'Personal'
        ELSE 'Otro'
    END as area
FROM empleados;

-- CASE Searched (condiciones)
SELECT nombre, salary,
    CASE 
        WHEN salary < 30000 THEN 'Bajo'
        WHEN salary < 60000 THEN 'Medio'
        ELSE 'Alto'
    END as nivel_salario
FROM empleados;

3. CAST - Conversión de Tipos

-- Convertir texto a fecha
SELECT CAST('2024-01-15' AS DATE);

-- Convertir número a texto
SELECT CAST(123.45 AS VARCHAR(10));

-- Sintaxis alternativa: ::
SELECT '2024-01-15'::DATE;
SELECT 123.45::NUMERIC;

📝 Resumen: COALESCE maneja valores nulos proporcionando alternativas, NULLIF evita errores de división por cero, CASE permite lógica condicional en consultas, y CAST convierte entre tipos de datos.

EJERCICIO RESUELTO

Mostrar nombre, salary y categoría: < 30000='Bajo', 30000-60000='Medio', > 60000='Alto'

SELECT nombre, salary,
    CASE 
        WHEN salary < 30000 THEN 'Bajo'
        WHEN salary >= 30000 AND salary < 60000 THEN 'Medio'
        ELSE 'Alto'
    END as categoria
FROM empleados;
