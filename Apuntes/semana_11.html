5. Ciclos - LOOP

-- Loop bÃ¡sico
CREATE OR REPLACE FUNCTION contar_hasta(n INTEGER)
RETURNS TEXT AS $$
DECLARE
    i INTEGER := 0;
    resultado TEXT := '';
BEGIN
    LOOP
        i := i + 1;
        resultado := resultado || i || ' ';
        IF i >= n THEN
            EXIT;
        END IF;
    END LOOP;
    RETURN resultado;
END;
$$ LANGUAGE plpgsql;

-- WHILE
WHILE i < n LOOP
    i := i + 1;
END LOOP;

-- FOR (rango)
FOR i IN 1..10 LOOP
    resultado := resultado || i;
END LOOP;
