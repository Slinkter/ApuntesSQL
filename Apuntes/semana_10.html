1. Vistas (CREATE VIEW)

-- Crear vista simple
CREATE VIEW vista_empleados_ventas AS
SELECT id, nombre, salary
FROM empleados
WHERE departamento = 'Ventas';

-- Usar la vista
SELECT * FROM vista_empleados_ventas;

-- Vista con JOIN
CREATE VIEW vista_resumen_pedidos AS
SELECT c.nombre as cliente, COUNT(p.id) as total_pedidos, SUM(p.total) as total
FROM clientes c
LEFT JOIN pedidos p ON c.id = p.id_cliente
GROUP BY c.nombre;

-- Eliminar vista
DROP VIEW vista_empleados_ventas;

3. CTEs Recursivas

-- Recorrer jerarquÃ­a de empleados
WITH RECURSIVE jerarquia AS (
    -- Caso base: CEO (sin manager)
    SELECT id, nombre, manager_id, 1 as nivel
    FROM empleados WHERE manager_id IS NULL
    
    UNION ALL
    
    -- Caso recursivo
    SELECT e.id, e.nombre, e.manager_id, j.nivel + 1
    FROM empleados e
    JOIN jerarquia j ON e.manager_id = j.id
)
SELECT * FROM jerarquia;
